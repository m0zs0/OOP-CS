```c#
public static T GetMax<T>(T a, T b) where T : IComparable<T>
{
    return a.CompareTo(b) > 0 ? a : b;
}

int maxInt = GetMax<int>(3, 5);
Console.WriteLine(maxInt)
string maxString = GetMax<string>("apple", "orange");
Console.WriteLine(maxString)
```
